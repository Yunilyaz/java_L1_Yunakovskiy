public class Example {
    public static void main(String[] args) {
        for (int i = 1; i <= 10; i++) {
            if (i % 2 == 0) {
                continue;  // Пропускает итерацию, если число четное. Оператор continue пропускает текущую итерацию и переходит к следующей
            }
            if (i == 5) {
                break;  // Прерывает выполнение цикла, если i равно 7. Оператор break завершает выполнение текущего цикла досрочно
            }
            
        }
    }
}
