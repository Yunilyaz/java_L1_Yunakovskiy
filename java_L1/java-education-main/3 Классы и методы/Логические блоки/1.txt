Вывод программы:
static logic // первый - статический блок выполняется один раз при первой загрузке класса
logic (1) id= 0 // второй - объект ещё не инициализирован (поле id имеет значение по умолчанию — 0)
logic (2) id= 1 // третий - после того, как поле id было инициализировано значением 1 (до выполнения конструктора)
ctor id= 100 // четвертый - в самом конце, когда конструктор присваивает новое значение полю id — 100
